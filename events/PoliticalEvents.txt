add_namespace = political

#Strasserist Coup
country_event = {
	id = political.1
	title = political.1.t
	desc = political.1.d
	picture = GFX_report_event_coup
	
	trigger = {
		strasserist > 0.5
		NOT = {
			has_government = strasserist
		}
		is_puppet = no
	}
	
	mean_time_to_happen = {
		days = 50
		modifier = {
			factor = 0.5
			strasserist > 0.55
		}	
	}
	
	option = {
		name = political.1.o1
		ai_chance = {
			factor = 100
		}
		set_politics = {
			ruling_party = strasserist
			elections_allowed = no
		}
		if = {
		    limit = { has_idea = weak_advisory_council }
			remove_ideas = weak_advisory_council
		}
		if = {
		    limit = { has_idea = strong_advisory_council }
			remove_ideas = strong_advisory_council
		}
	}
	
	option = {
		name = political.1.o2
		ai_chance = {
			factor = 0
		}
		start_civil_war = {
			ideology = strasserist
			size = 0.5
		}
	}
}
